><div class="dropdown-wrap">
    <div class="dropdown-onChange">click me</div>
    <ul class="onChange">
        <li data-category="sport">items-1</li>
        <li data-category="news">items-2</li>
        <li data-category="info">items-3</li>
    </ul>
</div>
<script>
    $(document).ready(function() {
        $('.dropdown-onChange').asDropdown({
            panel: '.onChange',
            data: 'category',
            onChange: function(value) {
                console.log('selected ' + value);
            }
        });
    });
</script>
